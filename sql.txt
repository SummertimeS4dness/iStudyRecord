
CREATE TABLE "SUBJECTS" (
	"SUBJECTSHORTNAME" VARCHAR2 (20) NOT NULL ,
	"SUBJECTFULLNAME" VARCHAR2 (20) NOT NULL ,
PRIMARY KEY ("SUBJECTSHORTNAME")
)
/

CREATE TABLE "STUDENTS" (
	"STUDENTLOGIN" VARCHAR2 (20) NOT NULL ,
	"STUDENTPASSWORD" VARCHAR2 (20) NOT NULL ,
	"STUDENTNAME" VARCHAR2 (20) NOT NULL ,
	"STUDENTGGROUP" VARCHAR2 (20) NOT NULL ,
	"ISSTAROSTA" VARCHAR2 (1) DEFAULT NO,
	"STAROSTAID" VARCHAR2(30) REFERENCES STUDENTS (STUDENTLOGIN),
PRIMARY KEY ("STUDENTLOGIN")
)
/

CREATE TABLE "LECTURERS" (
	"LECTURERLOGIN" VARCHAR2 (20) NOT NULL ,
	"LECTURERPASSWORD" VARCHAR2 (20) NOT NULL ,
	"LECTURERNAME" VARCHAR2 (20),
PRIMARY KEY ("LECTURERLOGIN")
)
/

CREATE TABLE "MARKS" (
	"SCORE" VARCHAR2 (20),
	"STUDENTLOGIN" VARCHAR2 (20) NOT NULL ,
	"SUBJECTSHORTNAME" VARCHAR2 (20) NOT NULL ,
	"LOGIN" VARCHAR2 (20) NOT NULL ,
	"CLASSDATE" VARCHAR2 (20) NOT NULL ,
PRIMARY KEY ("STUDENTLOGIN","SUBJECTSHORTNAME","LOGIN","CLASSDATE")
)
/

CREATE TABLE "CLASSES" (
	"CLASSDATE" VARCHAR2 (20) NOT NULL ,
	"SUBJECTSHORTNAME" VARCHAR2 (20) NOT NULL ,
	"LECTURERLOGIN" VARCHAR2 (20) NOT NULL ,
PRIMARY KEY ("CLASSDATE","SUBJECTSHORTNAME","LECTURERLOGIN")
)
/

CREATE TABLE "STUDENTSUBJECTLIST" (
	"SUBJECTSHORTNAME" VARCHAR2 (20) NOT NULL ,
	"STUDENTLOGIN" VARCHAR2 (20) NOT NULL ,
PRIMARY KEY ("SUBJECTSHORTNAME","STUDENTLOGIN")
)
/

CREATE TABLE "LECTURERSUBJECTLISTS" (
	"SUBJECTSHORTNAME" VARCHAR2 (20) NOT NULL ,
	"LECTURERLOGIN" VARCHAR2 (20) NOT NULL ,
PRIMARY KEY ("SUBJECTSHORTNAME","LECTURERLOGIN")
)
